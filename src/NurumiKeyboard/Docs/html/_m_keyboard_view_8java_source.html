632  private void checkDirection(PtIdLinkedWithPtIndex pp, PointF pt)
633  {
634  CircleLinkedWithPtId cp = new CircleLinkedWithPtId();
635  for(CircleLinkedWithPtId clpElement : clp) { // Find cp which has same pointerID of pp.
636  if(pp.pointerId == clpElement.pointerId) { // This is for link up circle number and pointer index.
637  cp = clpElement;
638  break;
639  }
640  }
641 
642  PointF oldPt = new PointF();
643  int circleNum = INVALID_CIRCLE;
644  for(PointF oldPtElement : oldPtArr) { // Find the started point with circle number.
645  circleNum = checkTouchedCircle((int)oldPtElement.x, (int)oldPtElement.y);
646  if(cp.circleNum == circleNum) {
647  oldPt = oldPtElement;
648  break;
649  }
650  }
651  if(circleNum == INVALID_CIRCLE) return;
652 
653  float distanceX = oldPt.x - pt.x;
654  float distanceY = oldPt.y - pt.y;
655 
656  if( Math.abs(distanceX) < swipeThreshold && Math.abs(distanceY) < swipeThreshold )
657  motion[circleNum-1] = DIRECTION_DOT;
658  else if( Math.abs(distanceX) > swipeThreshold || Math.abs(distanceY) > swipeThreshold ) {
659  if( Math.abs(distanceY/distanceX) < 1 && Math.abs(distanceX) > swipeThreshold) {
660  // Gradient is smaller than 1.
661  if(distanceX > 0)
662  motion[circleNum-1] = DIRECTION_LEFT;
663  else
664  motion[circleNum-1] = DIRECTION_RIGHT;
665  }
666  else if( Math.abs(distanceY/distanceX) >= 1 && Math.abs(distanceY) > swipeThreshold) {
667  // Gradient is 1 or larger.
668  if(distanceY > 0)
669  motion[circleNum-1] = DIRECTION_UP;
670  else
671  motion[circleNum-1] = DIRECTION_DOWN;
672  }
673  } // direction UDLR end
674  } // checkDirection fin
